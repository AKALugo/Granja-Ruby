def initialize (iden = 0, e = 0, sex = "", pe = 0, pi = nil, pa = nil, mo = nil)

    if (pi == :corto || pi == :grueso_y_robusto || pi == :ancho_y_aplanado)
        @pico = pi
    else
        return nil
    end

    if (pa == :anisodactilas || pa == :palmeadas)
        @patas = pa
    else 
        return nil
    end

    if (mo == :corredoras || mo == :voladoras || mo == :nadadoras)
        @movilidad = mo
    else
        return nil
    end

    super(iden ,e ,sex ,pe)
end
